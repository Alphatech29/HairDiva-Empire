CREATE TABLE hair_products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    product_name VARCHAR(100) NOT NULL,
    hair_type VARCHAR(50),
    description TEXT,
    color VARCHAR(50),
    tag VARCHAR(50),
    image_url VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE hair_variants (
    id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT NOT NULL,          -- link to main product
    length VARCHAR(50) NOT NULL,
    price DECIMAL(10,2) NOT NULL,     -- current price
    old_price DECIMAL(10,2),          -- old price (nullable)
    stock INT DEFAULT 0,
    sold INT DEFAULT 0,
    FOREIGN KEY (product_id) REFERENCES hair_products(id) ON DELETE CASCADE
);

CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_number VARCHAR(50) UNIQUE NOT NULL,       -- tracking number
    customer_name VARCHAR(100) NOT NULL,
    customer_email VARCHAR(100),
    customer_phone VARCHAR(20),

    products JSON NOT NULL,
    -- Example:
    -- [
    --   {
    --     "product_id": 1,
    --     "color": "Black",
    --     "length": "12",
    --     "quantity": 2,
    --     "unit_price": 5000
    --   },
    --   {
    --     "product_id": 2,
    --     "color": "Blonde",
    --     "length": "16",
    --     "quantity": 1,
    --     "unit_price": 5300
    --   }
    -- ]

    shipping_address TEXT NOT NULL,
    shipping_city VARCHAR(100) NOT NULL,
    shipping_state VARCHAR(100),

    subtotal DECIMAL(10,2) NOT NULL,                -- sum of item (qty Ã— unit_price)
    discount DECIMAL(10,2) DEFAULT 0.00,            -- discount amount
    vat_amount DECIMAL(10,2) NOT NULL,              -- tax
    total DECIMAL(10,2) NOT NULL,                   -- grand total

    status ENUM('pending','paid','shipped','completed','cancelled') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE admin (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
